//Вопрос# 2: Получить 2nd maximum salary

//Вопрос# 3: Вывести "EMPLOYEE NAME", DOB, "DEPARTMENT NAME" of "QA" DEPARTMENT, Sorting by "EMPLOYEE NAME" ASC

SELECT EMPNAME, DOB, DEPTNAME FROM DEPT_B INNER JOIN EMP_A ON DEPT_B.DEPTID=EMP_A.DEPTID WHERE DEPT_B.DEPTID=EMP_A.DEPTID ORDER BY EMPNAME ASC;

//Вопрос# 4: Вывести "EMPLOYEE NAME", "AGE", "DEPARTMENT NAME" Sort by "AGE" DESC

SELECT EMPNAME, TIMESTAMPDIFF(year, DOB, curdate()) AS AGE, DEPTNAME from DEPT_B INNER JOIN EMP_A ON DEPT_B.DEPTID=EMP_A.DEPTID WHERE DEPT_B.DEPTID=EMP_A.DEPTID order by AGE DESC;

//Вопрос# 5: Вывести "DEPARTMENT NAME", Number of Employees Sort by Number of Employees

SELECT EMPID, DEPTNAME from DEPT_B INNER JOIN EMP_A ON DEPT_B.DEPTID=EMP_A.DEPTID WHERE DEPT_B.DEPTID=EMP_A.DEPTID order by EMPID;

//Вопрос# 6: Вывести EMPLOYEE NAME, MANAGER NAME, DEPARTMENT NAME
         (показать всех работников включает тех, у кого нет менеджера)

SELECT EMPNAME, MGRID, DEPTNAME from DEPT_B INNER JOIN EMP_A ON DEPT_B.DEPTID=EMP_A.DEPTID WHERE DEPT_B.DEPTID=EMP_A.DEPTID;

//Вопрос# 7: Вывести EMPLOYEE NAME, MANAGER NAME, DEPARTMENT NAME
     (Из всех отделов где более одного работника)

SELECT EMPNAME, MGRID, DEPTNAME from DEPT_B INNER JOIN EMP_A ON DEPT_B.DEPTID=EMP_A.DEPTID WHERE DEPT_B.DEPTID=EMP_A.DEPTID AND MGRID LIMIT 1,2;
